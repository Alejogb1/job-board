// This is your Prisma schema file,
// learn more about it in the docs: https://pris.ly/d/prisma-schema

datasource db {
  provider     = "mysql"
  url          = env("DATABASE_URL")
  relationMode = "prisma"
}

generator client {
  provider = "prisma-client-js"
}

model CompanyImage {
  id            Int @id @default(autoincrement())
  company       Company @relation(fields: [company_id], references: [id])
  company_id    Int @unique  
  company_logo  String
}

model CompanyTeams {
  id          Int @id @default(autoincrement())
  company     Company? @relation(fields: [company_id], references: [id])
  company_id  Int @unique  
  title       String
}  

model Company {
  id                  Int @id @default(autoincrement())
  jobs                JobPost[]
  company_teams       CompanyTeams[]
  company_image       CompanyImage?
  company_name        String
  profile_description String?
  location            String
  company_webiste_url String
  created_at          DateTime
}

model JobLocation {
  id     Int    @id @default(autoincrement())
  job_id JobPost @relation(fields: [id], references: [id])
  state  String
  country String
  zip    String
}

model JobPost {
  id               Int @id @default(autoincrement())
  post_by_id       Int
  job_post_url     String @unique
  slug             String
  job_body         String
  company          Company? @relation(fields: [id], references: [id])
  job_location_id  JobLocation?
  tags             Tags[]
  is_active        Boolean
  is_remote        Boolean
  is_sponsored     Boolean
  created_at       DateTime
}

model Users {
  userId     Int      @id @default(autoincrement())
  username   String   @unique
  email      String   @unique
  created_at DateTime
}

model Tags {
  id         Int       @id @default(autoincrement())
  job_post_id JobPost  @relation(fields: [id], references: [id])
  created_at DateTime
  skillsets  Skillsets[]
  perks      Perks[]
}

model Skillsets {
  id       Int     @id @default(autoincrement())
  tag_id   Tags    @relation(fields: [id], references: [id])
  title    String
  created_at DateTime
}

model Perks {
  id        Int     @id @default(autoincrement())
  tag_id   Tags    @relation(fields: [id], references: [id])
  title     String
  created_at DateTime
}
